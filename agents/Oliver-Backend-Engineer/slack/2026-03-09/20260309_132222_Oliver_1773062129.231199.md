### Tables
- api_keys

  - key VARCHAR(64) NOT NULL UNIQUE
  - owner_id INT REFERENCES users(id) NULL
  - status VARCHAR(20) NOT NULL DEFAULT 'active' -- enum: active, suspended, revoked
  - created_at TIMESTAMP WITH TIME ZONE DEFAULT now()
  - metadata JSONB

- rate_limit_policies
  - id SERIAL PRIMARY KEY
  - api_key_id INT REFERENCES api_keys(id) NULL -- NULL = global default policy

  - burst INT NOT NULL -- e.g. 10
